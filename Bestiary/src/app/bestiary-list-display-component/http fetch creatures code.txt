//Loop through the JSON File to get all the creatures for each nature
    for (var key in this.natureJSON){
      for (var creatureKey in this.natureJSON[key]){
        const subscription = this.httpClient.get('https://bestiary-api.c-hypercube.ca/beast/' + creatureKey)
            .subscribe({
              next: (respData) => {
                this.creatureJSON = respData;
                dummyBeasts.push({species: key, name: creatureKey, rank: this.creatureJSON.rank, featurestraits: this.creatureJSON.features, 
                  drives: this.creatureJSON.drives, tactics: this.creatureJSON.tactics, desc: this.creatureJSON.description});
              }
          });
        this.destroyRef.onDestroy(() => {
          subscription.unsubscribe();
        });

        
      }  
    }
